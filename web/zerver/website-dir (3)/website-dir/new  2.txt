                function mOver ()
                {
                                start = new Date().getTime();
                
                timer = setInterval(function()
                {
                var walkingImg = document.getElementById("walk").src;
                if (walkingImg == localhost + "/images/walk1.png")
                {
                document.getElementById("walk").src = "/images/walk2.png";
                document.getElementById("walkSrc").innerHTML = walkingImg;
                }
                else if (walkingImg == localhost + "/images/walk2.png")
                {
                document.getElementById("walk").src = "/images/walk4.png";
                document.getElementById("walkSrc").innerHTML = walkingImg;
                }
                else if (walkingImg == localhost + "/images/walk3.png")
                {
                document.getElementById("walk").src = "/images/walk4.png";
                document.getElementById("walkSrc").innerHTML = walkingImg;
                }
                else if (walkingImg == localhost + "/images/walk4.png")
                {
                document.getElementById("walk").src = "/images/walk2.png";
                document.getElementById("walkSrc").innerHTML = walkingImg;
                }
                
                elapsed = 0;
                var time = new Date().getTime() - start;
                elapsed = Math.floor(time / 100) / 10;
                if(Math.round(elapsed) == elapsed) { elapsed += '.0'; }
                if (highscore < elapsed)
                    {
                    highscore = elapsed;
                    }
                    document.getElementById("scoreBoard").innerHTML = "elaped: " + elapsed + " Highscore: " + highscore;
                    }, 200);
                    }
                    
                    function moves()
                    {
                    count++;
                    document.getElementById("counts").innerHTML = "Count: " + count;
                    }
                    
                    function mOut()
                    {
                    document.getElementById("scoreBoard").innerHTML = "elaped: " + elapsed + " Highscore: " + highscore;
                    elapsed = 0.0;
                    start = 0.0;
                    clearInterval(timer);
                    document.getElementById("walk").src = "/images/walk1.png";
                    
                    }

                        Touchy.stopWindowBounce();
            
                        var touchMe = document.getElementById('touch-me');
            
                        Touchy(touchMe, true, function (hand, finger) {
                   if (hand.fingers.length > 1) {
                   return;
                   }
                   
                   finger.on('start', mOver);
                   finger.on('move', moves);
                   finger.on('end'  , mOut);
